5   import io.dropwizard.Application;
6   import io.dropwizard.assets.AssetsBundle;
7   import io.dropwizard.server.AbstractServerFactory;
8   import io.dropwizard.setup.Bootstrap;
9   import io.dropwizard.setup.Environment;
10  
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  
14  public class JMXProxyApplication extends Application<JMXProxyConfiguration> {
15      private static final Logger LOG = LoggerFactory.getLogger(JMXProxyApplication.class);
16  
17      public static void main(String[] args) throws Exception {
18          LOG.info("starting jmxproxy service");
19          new JMXProxyApplication().run(args);
20      }
21  
22      @Override
23      public String getName() {
24          return "jmxproxy";
25      }
26  
27      @Override
28      public void initialize(Bootstrap<JMXProxyConfiguration> bootstrap) {
29          bootstrap.addBundle(new AssetsBundle("/assets/", "/", "index.html"));
30      }
31  
32      @Override
33      public void run(JMXProxyConfiguration configuration, Environment environment) {
34          final ConnectionManager manager = new ConnectionManager(configuration.getApplicationConfiguration());
35          final JMXProxyResource resource = new JMXProxyResource(manager);
36          final JMXProxyHealthCheck healthCheck = new JMXProxyHealthCheck(manager);
37  
38          ((AbstractServerFactory) configuration.getServerFactory()).setJerseyRootPath("/jmxproxy/*");
39  
40          environment.lifecycle().manage(manager);
41          environment.jersey().register(resource);
42          environment.healthChecks().register("manager", healthCheck);
43      }
44  }
