27  public class Attribute implements JsonSerializable {
28      private static final Logger LOG = LoggerFactory.getLogger(Attribute.class);
29  
30      private Object attributeValue;
31  
32      public Attribute(Object attributeValue) {
33          this.attributeValue = attributeValue;
34      }
35  
36      public Object getAttributeValue() {
37          return attributeValue;
38      }
39  
40      public void serialize(JsonGenerator jgen, SerializerProvider sp) throws IOException, JsonProcessingException {
41          buildJson(jgen, attributeValue);
42      }
43  
44      public void serializeWithType(JsonGenerator jgen, SerializerProvider sp, TypeSerializer ts) throws IOException, JsonProcessingException {
45          buildJson(jgen, attributeValue);
46      }
47  
48      private void buildJson(JsonGenerator jgen, Object objectValue) throws IOException, JsonProcessingException {
49          if (objectValue == null) {
50              jgen.writeNull();
51          } else if (objectValue instanceof Boolean) {
52              jgen.writeBoolean((Boolean)objectValue);
53          } else if (objectValue instanceof JsonNode) {
54              jgen.writeTree((JsonNode)objectValue);
55          } else if (objectValue.getClass().isArray()) {
56              jgen.writeStartArray();
57              int length = Array.getLength(objectValue);
58              for (int i = 0; i < length; i++) {
59                  buildJson(jgen, Array.get(objectValue, i));
60              }
61              jgen.writeEndArray();
62          } else if (objectValue instanceof Iterable) {
63              Iterable data = (Iterable) objectValue;
64              jgen.writeStartArray();
65              for (Object objectEntry : data) {
66                  buildJson(jgen, objectEntry);
67              }
68              jgen.writeEndArray();
69          } else if (objectValue instanceof TabularData) {
70              TabularData data = (TabularData) objectValue;
71              jgen.writeStartArray();
72              for (Object objectEntry : data.values()) {
73                  buildJson(jgen, objectEntry);
74              }
75              jgen.writeEndArray();
76          } else if (objectValue instanceof CompositeData) {
77              CompositeData data = (CompositeData) objectValue;
78              jgen.writeStartObject();
79              for (String objectEntry : data.getCompositeType().keySet()) {
80                  jgen.writeFieldName(objectEntry);
81                  buildJson(jgen, data.get(objectEntry));
82              }
83              jgen.writeEndObject();
84          } else if (objectValue instanceof Number) {
85              Double data = ((Number) objectValue).doubleValue();
86              if (data.isNaN()) {
87                  jgen.writeString("NaN");
88              } else if (data.isInfinite()) {
89                  jgen.writeString("Infinity");
90              } else {
91                  jgen.writeNumber(((Number)objectValue).toString());
92              }
93          } else {
94              try {
95                  String input = objectValue.toString();
96                  List<JsonNode> parts = new ArrayList<JsonNode>();
97                  ObjectMapper mapper = new ObjectMapper();
98                  JsonParser parser = mapper.getFactory().createParser(input);
99  
100                 for (parser.nextToken(); parser.hasCurrentToken(); parser.nextToken()) {
101                     JsonToken tok = parser.getCurrentToken();
102                     int pos = (int)parser.getTokenLocation().getCharOffset();
103                     if (tok == JsonToken.START_OBJECT || tok == JsonToken.START_ARRAY) {
104                         parser.skipChildren();
105                     }
106                     parts.add(mapper.readTree(input.substring(pos, (int)parser.getTokenLocation().getCharOffset() + parser.getTextLength() + (tok == JsonToken.VALUE_STRING ? 2 : 0))));
107                 }
108 
109                 if (parts.isEmpty()) {
110                     jgen.writeString("");
111                 } else if (parts.size() == 1) {
112                     buildJson(jgen, parts.get(0));
113                 } else {
114                     buildJson(jgen, parts);
115                 }
116             } catch (JsonParseException e) {
117                 jgen.writeString(objectValue.toString());
118             }
119         }
120     }
