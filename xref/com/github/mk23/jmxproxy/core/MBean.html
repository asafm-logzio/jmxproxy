3   import com.fasterxml.jackson.core.JsonGenerator;
4   import com.fasterxml.jackson.core.JsonProcessingException;
5   import com.fasterxml.jackson.databind.JsonSerializable;
6   import com.fasterxml.jackson.databind.SerializerProvider;
7   import com.fasterxml.jackson.databind.jsontype.TypeSerializer;
8   
9   import java.io.IOException;
10  
11  import java.util.HashMap;
12  import java.util.Map;
13  import java.util.Set;
14  
15  public class MBean implements JsonSerializable {
16      private Map<String, History> attributes;
17      private ThreadLocal<Integer> limit = new ThreadLocal<Integer>() {
18          @Override protected Integer initialValue() {
19              return -1;
20          }
21      };
22  
23      public MBean() {
24          attributes = new HashMap<String, History>();
25      }
26  
27      public History addHistory(String attributeName, int size) {
28          if (!attributes.containsKey(attributeName)) {
29              History history = new History(size);
30              attributes.put(attributeName, history);
31          }
32  
33          return attributes.get(attributeName);
34      }
35  
36      public MBean setLimit(Integer limit) {
37          this.limit.set(limit);
38          return this;
39      }
40  
41      public Set<String> getAttributes() {
42          return attributes.keySet();
43      }
44  
45      public Attribute getAttribute(String attribute) {
46          History history = attributes.get(attribute);
47          if (history == null) {
48              return null;
49          }
50  
51          return history.getAttribute();
52      }
53  
54      public Attribute[] getAttributes(String attribute, int limit) {
55          History history = attributes.get(attribute);
56          if (history == null) {
57              return new Attribute[0];
58          }
59  
60          return history.getAttributes(limit);
61      }
62  
63      public void serialize(JsonGenerator jgen, SerializerProvider sp) throws IOException, JsonProcessingException {
64          buildJson(jgen);
65      }
66  
67      public void serializeWithType(JsonGenerator jgen, SerializerProvider sp, TypeSerializer ts) throws IOException, JsonProcessingException {
68          buildJson(jgen);
69      }
70  
71      private void buildJson(JsonGenerator jgen) throws IOException, JsonProcessingException {
72          int limit = this.limit.get();
73  
74          jgen.writeStartObject();
75          for (Map.Entry<String, History>attributeEntry : attributes.entrySet()) {
76              if (limit < 0) {
77                  jgen.writeObjectField(attributeEntry.getKey(), attributeEntry.getValue().getAttribute());
78              } else {
79                  jgen.writeObjectField(attributeEntry.getKey(), attributeEntry.getValue().getAttributes(limit));
80              }
81          }
82          jgen.writeEndObject();
83      }
84  }
