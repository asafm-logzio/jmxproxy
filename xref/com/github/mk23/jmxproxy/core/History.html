3   import org.slf4j.Logger;
4   import org.slf4j.LoggerFactory;
5   
6   public class History {
7       private static final Logger LOG = LoggerFactory.getLogger(History.class);
8   
9       private Attribute[] attributes;
10      private int current;
11  
12      public History(int size) {
13          attributes = new Attribute[size];
14      }
15  
16      public void addAttributeValue(Object attributeValue) {
17          Attribute attribute = new Attribute(attributeValue);
18          attributes[++current % attributes.length] = attribute;
19          if (current == Integer.MAX_VALUE) {
20              current = current % attributes.length + attributes.length;
21          }
22      }
23  
24      public Attribute getAttribute() {
25          return attributes[current % attributes.length];
26      }
27  
28      public Attribute[] getAttributes() {
29          return getAttributes(attributes.length);
30      }
31  
32      public Attribute[] getAttributes(int limit) {
33          int size = Math.min(limit != 0 ? limit : Integer.MAX_VALUE, Math.min(current, attributes.length));
34          Attribute[] rval = new Attribute[size];
35  
36          for (int i = 0; i < size; i++) {
37              rval[i] = attributes[(current - i) % attributes.length];
38          }
39  
40          return rval;
41      }
42  }
